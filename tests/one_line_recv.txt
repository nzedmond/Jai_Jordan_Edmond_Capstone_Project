python3 -c "
import socket, struct
s = socket.socket()
s.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
s.bind(('', 9000)); s.listen(1)
print('Waiting for connection...')
conn, addr = s.accept(); print(f'Connected: {addr}')
count = 0
while True:
    hdr = b''
    while len(hdr) < 12:
        chunk = conn.recv(12 - len(hdr))
        if not chunk: exit('Connection closed')
        hdr += chunk
    ts_ms, length = struct.unpack('>QI', hdr)
    body = b''
    while len(body) < length:
        body += conn.recv(length - len(body))
    count += 1
    print(f'frame {count:4d}  ts={ts_ms}  jpeg_len={length:,} bytes')
"
